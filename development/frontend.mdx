## Application initialization

The main entry point is `src/main.js`:

```javascript
import './assets/main.css'
import { createApp } from 'vue'
import { createPinia } from 'pinia'
import { themeConfig } from './theme'
import PrimeVue from 'primevue/config'
import ToastService from 'primevue/toastservice'
import ConfirmationService from 'primevue/confirmationservice'
import Tooltip from 'primevue/tooltip'
import App from './App.vue'
import router from './router'

const app = createApp(App)

app.use(createPinia())
app.use(PrimeVue, themeConfig)
app.use(ToastService)
app.use(ConfirmationService)
app.directive('tooltip', Tooltip)
app.use(router)

app.mount('#app')
```

```javascript
// stores/accountStore.js

export const useAccountStore = defineStore('account', {
  state: () => ({
    accounts: [],
    categories: [],
    actions: []
  }),
  
  getters: {
    totalIncome: (state) => {
      // Calculate total income
    },
    totalExpense: (state) => {
      // Calculate total expense
    }
  },
  
  actions: {
    async fetchAccounts() {
      // Fetch accounts from API
    },
    async addAccount(accountData) {
      // Add new account
    }
  }
})
```

<Note>
The store handles all API communication and provides reactive data to components.
</Note>

## Routing configuration

The application uses Vue Router with hash-based history:

```javascript
import { createRouter, createWebHashHistory } from 'vue-router'
import ChartView from '@/views/ChartView.vue'

const router = createRouter({
  history: createWebHashHistory(import.meta.env.BASE_URL),
  routes: [
    {
      path: '/',
      name: 'chart',
      component: ChartView
    },
    {
      path: '/list',
      name: 'list',
      component: () => import('@/views/ListView.vue')
    },
    {
      path: '/add',
      name: 'add',
      component: () => import('@/views/AddView.vue')
    }
  ]
})

export default router
```

<Info>
Hash-based routing (`createWebHashHistory`) is used for better compatibility when serving the app as a single HTML file.
</Info>

## Vite configuration

The project uses custom Vite configuration:

```javascript
import { fileURLToPath, URL } from 'node:url'
import { defineConfig } from 'vite'
import { viteSingleFile } from 'vite-plugin-singlefile'
import vue from '@vitejs/plugin-vue'
import vueDevTools from 'vite-plugin-vue-devtools'
import tailwindcss from '@tailwindcss/vite'

export default defineConfig({
  plugins: [
    vue(),
    vueDevTools(),
    tailwindcss(),
    viteSingleFile()
  ],
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url))
    }
  },
  base: './',
  build: {
    minify: false // Readable code for debugging
  }
})
```

### Key features

- **Path alias**: `@` maps to `src/` directory
- **Relative base**: Allows opening HTML directly in browser
- **Single file build**: Bundles everything into one HTML file
- **Unminified output**: Easier debugging of production builds
- **Vue DevTools**: Development debugging tools

## API integration

API calls are made using Axios with a configured base URL:

```javascript
import axios from 'axios'

const api = axios.create({
  baseURL: import.meta.env.VITE_API_URL || 'http://localhost:8080/api',
  headers: {
    'Content-Type': 'application/json'
  }
})

// Example: Fetch all account books
export const getAccountBooks = () => api.get('/account-books')

// Example: Create new account book
export const createAccountBook = (data) => api.post('/account-books', data)
```

## Best practices

<Info>
Follow Vue.js 3 Composition API patterns for better code organization and reusability.
</Info>

### Component composition

Use composables for shared logic:

```javascript
// composables/useAccountData.js
import { ref, computed } from 'vue'
import { useAccountStore } from '@/stores/accountStore'

export function useAccountData() {
  const store = useAccountStore()
  const loading = ref(false)
  
  const totalBalance = computed(() => {
    return store.totalIncome - store.totalExpense
  })
  
  const fetchData = async () => {
    loading.value = true
    await store.fetchAccounts()
    loading.value = false
  }
  
  return { totalBalance, loading, fetchData }
}
```

### Error handling

Implement proper error handling for API calls:

```javascript
import { useToast } from 'primevue/usetoast'

const toast = useToast()

try {
  await createAccountBook(formData)
  toast.add({
    severity: 'success',
    summary: 'Success',
    detail: 'Transaction added successfully',
    life: 3000
  })
} catch (error) {
  toast.add({
    severity: 'error',
    summary: 'Error',
    detail: error.message || 'Failed to add transaction',
    life: 3000
  })
}
```
